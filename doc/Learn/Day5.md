:: ผลคูณคาทีเซียน ::

--> คู่อันดับ (x,y),(1,2),(x,2),(เสื้อ,กางเกง)
--> การเท่ากันของคู่อันดับ
    (a,b) = (c,d) 
    a = c && b = d
    ex 1 : ถ้า x และ y เป็นจำนวนจริงที่ (x+y,2) = (6,x-y)
    แล้ว x และ y เป็นเท่าไหร่ ? 

    x+y = 6    |  x+y = 6
    2   = x-y  |  x-y = 2 

    2x = 8 ; x = 4
    4 - y = 2 ; y = 2 

    = (x,y) = (4,2) 

--> ผลคูณคาทีเซียน
    A x B = {(a,b) | a E A & b E B}
    --> new Set(a,b) ==> a from A,b from B

    A = {1,2,3} || B = {a,b,c,d}
    A x B = {(1,a),(1,b),(1,c),(1,d),
             (2,a),(2,b),(2,c),(2,d),
             (3,a),(3,b),(3,c),(3,d),
             (4,a),(4,b),(4,c),(4,d)    
    }
